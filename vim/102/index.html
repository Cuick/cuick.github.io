if !exists(":DiffOrig")
    command DiffOrig vert new | set bt=nofile | r ++edit # | 0d_ | diffthis | wincmd p | diffthis
endif

syntax on

set t_Co=256
set mouse=a


set expandtab
set sw=4
set tabstop=4
set softtabstop=4

set nocompatible
set rtp+=~/.vim/bundle/Vundle.vim

set helplang=cn
set nu
set cindent
set smartindent
set showmatch
set ignorecase smartcase
set nowrapscan
set incsearch
set hlsearch
set magic
set sm
set backspace=indent,eol,start
set cmdheight=1

set laststatus=2
set autoread
set list lcs=tab:>-,trail:-

set fileencodings=utf-8,gb2312,gbk,gb18030
set termencoding=utf-8
set fileformats=unix
set encoding=utf-8

call vundle#begin()
